CurveRaycastUtils - Documentation

Description

CurveRaycastUtils est une classe utilitaire permettant d'effectuer un raycast courbé dans Unity. Cela est utile pour simuler des trajectoires de projectiles, des tirs paraboliques ou des effets visuels.

Installation

Ajoutez le fichier CurveRaycastUtils.cs à votre projet Unity.

Utilisation

Effectuer un raycast courbé et tracer la trajectoire :

bool collision = CurveRaycastUtils.EffectuerRaycastEtTracer(
    pointDepart, pointArrivee, directionCourbe, facteurCourbe, pas, layerCollision,
    out RaycastHit hit, couleurCourbe, couleurCollision, duration, depthTest
);

Exemple d'utilisation

using UnityEngine;
public class ExampleCurveRaycast : MonoBehaviour
{
    public Vector3 pointDepart;
    public Vector3 pointArrivee;
    public Vector3 directionCourbe = Vector3.up;
    public float facteurCourbe = 5f;
    public float pas = 0.1f;
    public LayerMask layerCollision;
    public Color couleurCourbe = Color.red;
    public Color couleurCollision = Color.green;
    public float duration = 1.0f;
    public bool depthTest = true;
    
    private void OnDrawGizmos()
    {
        if (CurveRaycastUtils.EffectuerRaycastEtTracer(
            pointDepart, pointArrivee, directionCourbe, facteurCourbe, pas, layerCollision,
            out RaycastHit hit, couleurCourbe, couleurCollision, duration, depthTest))
        {
            Debug.Log($"Collision détectée avec : {hit.collider.name} à la position {hit.point}");
        }
        else
        {
            Debug.Log("Aucune collision détectée.");
        }
    }
}
